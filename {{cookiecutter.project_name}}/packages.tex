%

%
% Usage:
%
%\begin{equation} \label{eq:1}
%   Y_i = \beta_0 + \beta_1 X_i + \epsilon_i
% \end{equation}
%
\usepackage{amsmath}

% ===  array ===
%
% Improve array and tabular formatting
%
\usepackage{array}

% === authblk ===
%
% Support for footnote style author/affiliation
%
\usepackage{authblk}

% === booktabs ===
%
% Support for better table formatting.
%
\usepackage{booktabs}

% === caption ===
% Support for captions.
%
% Usage:
%
% \captionsetup[FLOAT_TYPE]{OPTIONS}
%
\usepackage{caption}
\captionsetup[lstlisting]{
	justification=raggedright,
	singlelinecheck=false
}

% === listings ===
%
% Support for source code listings.
%
% Usage:
%
% \lstinputlisting[language=Python]{source_filename.py}
% \lstinputlisting[language=Python, firstline=37, lastline=45]{source_filename.py}
%
\usepackage{listings}
\lstset{
	frame=single,
	basicstyle=\small,
	breaklines=true,
	columns=fullflexible,
	aboveskip=10pt,
	belowskip=10pt,
	framesep=0.1in,
	xleftmargin=0.1in,
	xrightmargin=0.1in
}
